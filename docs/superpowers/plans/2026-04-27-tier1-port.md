- [ ] **Step 4: Apply backend port**

Add the in-memory composer tracker. Critical: use a mutex around the map. TTL-expire entries after, say, 30 seconds so a disconnected agent's "composing" status doesn't linger forever.

```go
// internal/conversation/ws.go (anticipated; adapt to actual diff)
var (
    composersMu sync.Mutex
    composers   = make(map[string]map[int]time.Time) // convUUID -> userID -> lastSeen
)

const composerTTL = 30 * time.Second

func (m *Manager) BroadcastComposing(convUUID string, userID int, isComposing bool) {
    composersMu.Lock()
    if _, ok := composers[convUUID]; !ok {
        composers[convUUID] = make(map[int]time.Time)
    }
    if isComposing {
        composers[convUUID][userID] = time.Now()
    } else {
        delete(composers[convUUID], userID)
    }
    // GC expired
    for uid, t := range composers[convUUID] {
        if time.Since(t) > composerTTL {
            delete(composers[convUUID], uid)
        }
    }
    activeUserIDs := make([]int, 0, len(composers[convUUID]))
    for uid := range composers[convUUID] {
        activeUserIDs = append(activeUserIDs, uid)
    }
    composersMu.Unlock()

    m.broadcastToUsers([]int{}, wsmodels.Message{
        Type: wsmodels.MessageTypeComposing,
        Data: map[string]any{
            "conversation_uuid": convUUID,
            "user_ids":          activeUserIDs,
        },
    })
}
```

Add the `MessageTypeComposing` constant in `internal/ws/models/`. Wire the inbound WS message routing in `internal/ws/hub.go` to call `BroadcastComposing` when a client sends `{"type":"composing","conversation_uuid":"...","is_composing":true}`.

// internal/migrations/v2.1.3.go
package migrations

import (
    "github.com/jmoiron/sqlx"
    "github.com/knadh/koanf/v2"
    "github.com/knadh/stuffbin"
)

// V2_1_3 adds ticket merging support: the merged_into_conversation_id
// column tracks which primary a merged-secondary points to.
func V2_1_3(db *sqlx.DB, fs stuffbin.FileSystem, ko *koanf.Koanf) error {
    stmts := []string{
        `ALTER TABLE conversations ADD COLUMN IF NOT EXISTS merged_into_conversation_id BIGINT
            REFERENCES conversations(id) ON DELETE SET NULL`,
        `CREATE INDEX IF NOT EXISTS index_conversations_merged_into
            ON conversations (merged_into_conversation_id)
            WHERE merged_into_conversation_id IS NOT NULL`,
    }
    for _, q := range stmts {
        if _, err := db.Exec(q); err != nil {
            return err
        }
    }
    return nil
}
